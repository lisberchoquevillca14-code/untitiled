  from {
    transform: rotate(0deg) translateX(150px) rotate(0deg);
  }
  to {
    transform: rotate(360deg) translateX(150px) rotate(-360deg);
  }
}

<script>
const words = [
"Mi amor", "Mi cielo", "Te amo", "Eres todo",
"Mi vida", "Mi razón", "Mi luz", "Mi estrella",
"Contigo", "Siempre", "Abrazo", "Cariño",
"Mi destino", "Felicidad", "Sueños", "Inspiración"
];

const container = document.getElementById("container");

words.forEach((word, i) => {
  let span = document.createElement("span");
  span.className = "word";
  span.innerText = word;

  let angle = (360 / words.length) * i;

  span.style.top = "50%";
  span.style.left = "50%";
  span.style.transform = `rotate(${angle}deg) translateX(150px)`;

  span.style.animationDuration = (10 + Math.random() * 10) + "s";

  container.appendChild(span);
});
</script>
